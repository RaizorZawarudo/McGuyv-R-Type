40 void update(std::vector<EntityID> &allEntities) override {
41 for (EntityID ent : EntityViewer<Position, Velocity, EntityModelType, PitchYawRoll, Collider>(*_em.get())) {
42 _ent = ent;
43 EntityModelType* entityType = _em->Get<EntityModelType>(ent);
44 Position* entityPos = _em->Get<Position>(ent);
45 PitchYawRoll* pitchYawRoll = _em->Get<PitchYawRoll>(ent);
46 ModelDimensions* modelDimensions = _em->Get<ModelDimensions>(ent);
47 Collider* entityCollider = _em->Get<Collider>(ent);
48 Velocity* entityVel = _em->Get<Velocity>(ent);
49 Owner *entityOwner = _em->Get<Owner>(ent);
50 int forward = 0;
51
52 if (entityType->modelType == RL::ModelType::SPACESHIP) {
53 Input* entityMovement = _em->Get<Input>(ent);
54 for (int keypressed : entityMovement->_inputQueue) {
55 if (keypressed == UP || keypressed == UP2) moveUp(entityPos, entityVel, entityMovement, pitchYawRoll);
56 else if (keypressed == DOWN) moveDown(entityPos, entityVel, entityMovement, pitchYawRoll);
57 else {
58 if (pitchYawRoll->pitch > 0.0f) pitchYawRoll->pitch -= 0.5f;
59 else if (pitchYawRoll->pitch < 0.0f) pitchYawRoll->pitch += 0.5f;
60 }
61
62 if (keypressed == LEFT || keypressed == LEFT2) moveLeft(entityPos, entityVel, entityMovement, pitchYawRoll);
63 else if (keypressed == RIGHT) moveRight(entityPos, entityVel, entityMovement, pitchYawRoll);
64 else {
65 if (pitchYawRoll->roll > 0.0f) pitchYawRoll->roll -= 0.5f;
66 else if (pitchYawRoll->roll < 0.0f) pitchYawRoll->roll += 0.5f;
67 }
68
69 if (keypressed == FORWARD) forward = 1;
70 }
71 if (forward == 1 )moveForward(entityPos, entityVel, entityMovement, pitchYawRoll);
72 else
73 if (entityPos->pos.z > PLAYERSTARTINGZ) entityPos->pos.z -= 0.2f;
74 forward = 0;
75 }
76 if (entityType->modelType == RL::ModelType::ENNEMY) {
77 Input* entityMovement = _em->Get<Input>(ent);
78 for (int keypressed : entityMovement->_inputQueue) {
79 if (keypressed == UP || keypressed == UP2) moveUp(entityPos, entityVel, entityMovement, pitchYawRoll);
80 else if (keypressed == DOWN) moveDown(entityPos, entityVel, entityMovement, pitchYawRoll);
81 else {
82 if (pitchYawRoll->pitch > 0.0f) pitchYawRoll->pitch -= 0.5f;
83 else if (pitchYawRoll->pitch < 0.0f) pitchYawRoll->pitch += 0.5f;
84 }
85
86 if (keypressed == LEFT || keypressed == LEFT2) moveLeft(entityPos, entityVel, entityMovement, pitchYawRoll);
87 else if (keypressed == RIGHT) moveRight(entityPos, entityVel, entityMovement, pitchYawRoll);
88 else {
89 if (pitchYawRoll->roll > 0.0f) pitchYawRoll->roll -= 0.5f;
90 else if (pitchYawRoll->roll < 0.0f) pitchYawRoll->roll += 0.5f;
91 }
92 }
93 if (_assetManager->getMaps().at(_assetManager->getCurrentMapBeingPlayed())->getCurrentSpeed() != _assetManager->getMaps().at(_assetManager->getCurrentMapBeingPlayed())->getScrollSpeed())
94 entityVel->z = _assetManager->getMaps().at(_assetManager->getCurrentMapBeingPlayed())->getCurrentSpeed();
95 entityPos->pos.z += entityVel->z * (entityOwner->ownerType) * (-1);
96 }
97 if (entityType->modelType == RL::ModelType::PROJECTILE) {
98 entityPos->pos.x += entityVel->x;
99 entityPos->pos.y += entityVel->y * (-1);
100 entityPos->pos.z += entityVel->z * (entityOwner->ownerType) * (-1);
101 }
102 if (entityType->modelType == RL::ModelType::OBSTACLE) {
103 //if we spawned the boss but are rushing towards it, the obstacles must speed up as well. Keep the 2 final waves of a level easy
104 if (_assetManager->getMaps().at(_assetManager->getCurrentMapBeingPlayed())->getCurrentSpeed() != _assetManager->getMaps().at(_assetManager->getCurrentMapBeingPlayed())->getScrollSpeed())
105 entityVel->z = _assetManager->getMaps().at(_assetManager->getCurrentMapBeingPlayed())->getCurrentSpeed();
106 entityPos->pos.z += entityVel->z * (entityOwner->ownerType) * (-1);
107 }
108 if (entityType->modelType == RL::ModelType::POWERUP) {
109 pitchYawRoll->yaw += 0.5f;
110 entityPos->pos.z += entityVel->z * (-1);
111 }
112 }

156 {
157 pitchYawRoll->roll > -20.0f ? pitchYawRoll->roll -= 1.0f : pitchYawRoll->roll -= 0.0f;
158
159 if (pos->pos.x > MAXLEFT) return;
